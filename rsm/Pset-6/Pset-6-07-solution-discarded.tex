\begin{alignat*}{2}
4369 
 & = 1 \mult 16^{3} + 1 \mult 16^{2} + 1 \mult 16^{1} + 1 \mult 16^{0}
 && = 1111_{16}
\\
 & = 1 \mult 8^{4} + 4 \mult 8^{2} + 2 \mult 8^{1} + 1 \mult 8^{0}
 && = 10421_{8}
\\
 & = 1 \mult 4^{6} + 1 \mult 4^{4} + 1 \mult 4^{2} + 1 \mult 4^{0} 
 && = 1010101_{4}
\\
 & = 1 \mult 2^{12} + 1 \mult 2^{8} + 1 \mult 2^{4} + 1 \mult 2^{0}
 && = 1000100010001_{2}
\end{alignat*}
