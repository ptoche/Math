\item $\Frac{2\pi}{3}\left(10\sqrt{3}+3\right)$
\item $\Frac{20\pi}{3}$
\item $\Frac{20\sqrt{3}\pi}{3}$
\item $13\pi$
\item $20\pi$
\end{enumerate}
