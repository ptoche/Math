Let $(a_{n})$ be a number sequence defined as follows:
\begin{align*}
a_{1} = 1, \quad
a_{2} = 1, \quad 
a_{3} = \frac{1}{2} \quad
\text{and}\quad
a_{n+1} = \frac{a_{n}^{3}a_{n-2}}{a_{n-1}^{3}+a_{n-2}a_{n-1}a_{n}} 
\end{align*}
where $n=3,4,\ldots$

Calculate
\begin{align*}
S_{10} 
= \frac{a_{1}}{a_{2}} 
+ 2 \frac{a_{2}}{a_{3}} 
+ 3 \frac{a_{3}}{a_{4}} 
+ \ldots 
+ 10 \frac{a_{10}}{a_{11}} 
\end{align*}

\begin{enumerate}
\item $11!$
\item $11!-1$
\item $10!+3$
\item $10!+11$
\item $10 \cdot 11$
\end{enumerate}
