\tikzset{% 
  myfig/.pic={%

      \coordinate (A) at (-1,0);

      \coordinate (D) at (3,0);

        {(intersection-1) circle [radius=0.8pt]};

      \node[below left] at (A) {$A$};
      \node[below right] at (B) {$B$};
      \node[below left] at (C) {$C$};
      \node[below right] at (D) {$D$};
      
      \node[above] at (intersection-1) {$E$};
